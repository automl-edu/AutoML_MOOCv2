\begin{frame}{Greedy Ensemble Selection}
Assume a set of trained models and a validation set.
\begin{itemize}
    \item Goal: optimize weighted averaging without continuous optimization
    \item Standard method in AutoML systems~\liturl{Purucker et al. (2023b)}{https://proceedings.mlr.press/v224/purucker23b.html}
\end{itemize}
\vspace{0.2cm}
\textbf{Algorithm~\liturl{Caruana et al. (2004)}{https://dl.acm.org/doi/10.1145/1015330.1015432}:}
\begin{enumerate}
    \item Start with an empty ensemble
    \item Add the model that gives the largest validation improvement
    \item Repeat until the iteration budget is reached
    \item Return the best intermediate ensemble
\end{enumerate}
\end{frame}

\begin{frame}{Greedy Ensemble Selection: Pseudocode}
\scriptsize
\textbf{Greedy Ensemble Selection (GES)}
\begin{algorithmic}[1]
\State \textbf{Input:} Base models $\{\bl[1],\ldots,\bl[M]\}$, validation set $\datasetval$, budget $T$
\State \textbf{Output:} Ensemble weights $\ensweights^\star$
\State $\vec{c} \gets (0,\ldots,0)$; $s \gets 0$; $b \gets +\infty$; $\vec{c}^{\,\star} \gets \vec{c}$
\State \textbf{for} $t \gets 1$ \textbf{to} $T$ \textbf{do}
\State \hspace{1em}\textbf{for} $m \gets 1$ \textbf{to} $M$ \textbf{do}
\State \hspace{2em}$s_m \gets \frac{t-1}{t} s + \frac{1}{t}\bl[m](\x)$
\State \hspace{2em}$r_m \gets \riske(s_m;\datasetval)$
\State \hspace{1em}\textbf{end for}
\State \hspace{1em}$m_t \gets \argmin_{m \in \{1,\ldots,M\}} r_m$
\State \hspace{1em}$s \gets s_{m_t}$; $c_{m_t} \gets c_{m_t} + 1$
\State \hspace{1em}\textbf{if} $r_{m_t} < b$ \textbf{then}
\State \hspace{2em}$b \gets r_{m_t}$; $\vec{c}^{\,\star} \gets \vec{c}$
\State \hspace{1em}\textbf{end if}
\State \textbf{end for}
\State \textbf{for} $m \gets 1$ \textbf{to} $M$ \textbf{do}
\State \hspace{1em}$\ensweights^\star_m \gets c^\star_m / \sum_{j=1}^{M} c^\star_j$
\State \textbf{end for}
\State \Return $\ensweights^\star$
\end{algorithmic}
\end{frame}

\begin{frame}{Why GES Works Well in Practice}
\begin{itemize}
    \item Greedy steps are simple and fast
    \item Naturally yields sparse ensembles
    \item Can handle arbitrary performance metrics
    \item Often competitive with more complex optimizers
\end{itemize}
\vspace{0.3cm}
The final weight of a model is proportional to how often it was selected.
\end{frame}

\begin{frame}{Tricks of the Trade}
 \begin{itemize}
  \item Use cross-validation instead of a single validation set (see videos on stacking)
  \item Use a secondary metric (log loss) for tie breaking
  \item Do not prune based on the validation set~\liturl{Purucker et al. (2023b)}{https://proceedings.mlr.press/v224/purucker23b.html}
  \item Prune the number of models in the library to only include the top 50 - top 200~\liturl{Caruana et al. (2004)}{https://dl.acm.org/doi/10.1145/1015330.1015432}
  \item When using multiple model classes, include models from each class nonetheless
  \item Insights can be applied to optimize weights with CMA-ES~\liturl{Purucker et al. (2023a)}{https://proceedings.mlr.press/v224/purucker23a.html}
 \end{itemize}

\end{frame}


\begin{frame}{Summary}
Most important insights from this video:
\begin{enumerate}
    \item GES greedily builds an ensemble by repeated model additions
    \item It is metric-agnostic, sparse, and practical for AutoML
    \item Weight optimization can overfit validation data, so strict evaluation is required
\end{enumerate}
\end{frame}
