\begin{frame}{From Single-Layer to Multilayer Stacking}
\begin{itemize}
    \item Single-layer stacking trains one meta-model on base predictions
    \item Multilayer stacking repeats this idea across several levels
\end{itemize}
\vspace{0.2cm}
\begin{enumerate}
    \item Train level-0 base models
    \item Train one or more level-1 models on level-0 predictions
    \item Train a final model on level-1 predictions
\end{enumerate}
\vspace{0.2cm}
In principle, this can be extended to deeper stacks.
\end{frame}

\begin{frame}{How to ``train'' Stacknet?}

\begin{itemize}
    \item Use out-of-fold predictions to train meta-model
    \item Architecture could be optimized, but was chosen manually
    \item Analogy of training a neural network
\end{itemize}

\begin{frame}{Selecting models}
\begin{itemize}
 \item Base models
 \begin{itemize}
  \item Manually~\liturl{Michialidis, 2017}{https://discovery.ucl.ac.uk/id/eprint/10031000/1/Full_copy.pdf}
  \item Evolutionary Search~\liturl{Chen et al. (2018)}{https://dl.acm.org/doi/abs/10.1145/3205455.3205586}
  \item Meta-Learning~\liturl{Salinas and Erickson (2024)}{https://proceedings.mlr.press/v256/salinas24a.html}
  \item Re-use model types from level 1 in all further levels~\liturl{Erickson et al. (2020)}{https://arxiv.org/abs/2003.06505}
 \end{itemize}
 \item Combiner
 \begin{itemize}
  \item Greedy Ensemble Selection~\liturl{Erickson et al. (2020)}{https://arxiv.org/abs/2003.06505}
  \item Logistic regression~\liturl{Michialidis, 2017}{https://discovery.ucl.ac.uk/id/eprint/10031000/1/Full_copy.pdf}
  \item Any other ML model
 \end{itemize}

\end{itemize}

\end{frame}

\begin{frame}{Summary}
Most important insights from this video:
\begin{enumerate}
    \item Multilayer stacking repeatedly learns combinations across levels
    \item Design choices for higher-level models strongly affect performance
    \item AutoGluon is a practical example of stacking plus greedy selection
\end{enumerate}
