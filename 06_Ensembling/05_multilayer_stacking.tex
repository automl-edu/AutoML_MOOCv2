\begin{frame}{From Single-Layer to Multilayer Stacking}
\begin{itemize}
    \item Single-layer stacking trains one meta-model on base predictions
    \item Multilayer stacking repeats this idea across several levels
\end{itemize}
\vspace{0.2cm}
\begin{enumerate}
    \item Train level-0 base models
    \item Train one or more level-1 models on level-0 predictions
    \item Train a final model on level-1 predictions
\end{enumerate}
\vspace{0.2cm}
In principle, this can be extended to deeper stacks.
\end{frame}

\begin{frame}{Design Choices in Higher Layers}
\begin{itemize}
    \item Typical higher-level models:
    \begin{itemize}
        \item Linear or logistic regression
        \item Greedy ensemble selection on previous-layer predictions
    \end{itemize}
    \item Use probabilities instead of hard labels for richer meta-features
    \item Decide whether to include original input features
\end{itemize}
\end{frame}
