\begin{frame}{What Is Stacking?}
\begin{columns}
    \begin{column}{0.52\textwidth}
        \begin{figure}
            \centering
            \includegraphics[width=0.95\textwidth]{06_Ensembling/images/stacking_better.png}
        \end{figure}
    \end{column}
    \begin{column}{0.48\textwidth}
        \begin{itemize}
            \item Level-0: train multiple base models
            \item Build meta-features from their predictions
            \item Level-1: train a meta-model on these features
            \item Also called stacked generalization~\liturl{Wolpert (1992)}{https://www.sciencedirect.com/science/article/abs/pii/S0893608005800231}
        \end{itemize}
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Single-Layer Stacking Procedure}
\begin{enumerate}
    \item Train base models $f_1,\dots,f_M$ on training data
    \item For each training instance, collect base-model predictions
    \item Construct meta-dataset with these predictions
    \item Train a meta-model on the meta-dataset
\end{enumerate}
\vspace{0.2cm}
\begin{itemize}
    \item Optional: include original features in the meta-dataset
    \item Difference to averaging: combination is learned, not fixed
\end{itemize}
\end{frame}

\begin{frame}{Cross-Validated Stacking}
\begin{columns}
    \begin{column}{0.5\textwidth}
        \begin{figure}
            \centering
            \includegraphics[width=0.55\textwidth]{06_Ensembling/images/stacking_cv_better.png}
        \end{figure}
    \end{column}
    \begin{column}{0.5\textwidth}
        \begin{itemize}
            \item Use out-of-fold predictions to train meta-model
            \item Reduces overfitting and leakage
            \item Usually generalizes better than naive in-sample stacking
        \end{itemize}
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Blending vs Cross-Validated Stacking}
\begin{itemize}
    \item \textbf{Blending}: use one holdout split for meta-features
    \item \textbf{CV stacking}: use out-of-fold predictions from $K$ folds
\end{itemize}
\vspace{0.2cm}
\begin{itemize}
    \item Blending is cheaper but uses less training data
    \item CV stacking is more expensive but usually more robust
\end{itemize}
\end{frame}

\begin{frame}{Higher Level Models}
\vfill
Popular choices of combiner models in stacking include:
\begin{itemize}
    \item linear / logistic regression
    %\item random forest (especially in the case of multilevel stacking)
    \item GES (average the predictions of the models at the previous level)
\end{itemize}
\vfill
Higher-level training data:
\begin{itemize}
    \item Probabilities instead of class labels~\liturl{Ting and Witten (1999)}{https://dl.acm.org/doi/10.5555/1622859.1622868}
    \item Include original dataset $\X$ or not?
\end{itemize}
\vfill
\end{frame}

\begin{frame}{Summary}
Most important insights from this video:
\begin{enumerate}
    \item Stacking learns how to combine base-model predictions
    \item Out-of-fold meta-features are important to avoid leakage
    \item Blending trades robustness for lower computational cost
\end{enumerate}
\end{frame}
