\begin{frame}{Motivation}
\begin{itemize}
    \item One-shot models remove the need to retrain individual architectures, but the \alert{search over subgraphs is still discrete}
    \item Discrete search prevents us from using \textbf{gradient descent} — a powerful optimisation tool in deep learning
    \item Idea: relax discrete operation choices to \alert{continuous weights} so that architectures become differentiable w.r.t. a loss
\end{itemize}
\pause
\vspace{0.5em}
\begin{colorblock}[black]{sintefgrey}{Key Question}
Can we formulate NAS as a \textbf{gradient-based optimisation} problem, optimizing over architecture parameters and network weights jointly?
\end{colorblock}
\end{frame}

\begin{itemize}
    \item Use one-shot model with continuous architecture weight $\alpha$ for each operator
    \begin{itemize}
        \item[] $x^{(j)} = \sum_{i<j}\tilde{o}^{(i,j)}(x^{(i)}) = \sum_{i<j}\sum_{o\in\mathcal{O}}
%			\text{softmax}(\alpha)
			\frac{exp(\alpha_{o}^{(i,j)})}{\sum_{o^{\prime}\in\mathcal{O}}exp(\alpha_{o^{\prime}}^{(i,j)})}o(x^{(i)})$
    \end{itemize}
\end{itemize}
\pause
    \begin{itemize}
         \item By optimizing architecture weights $\alpha$, DARTS assigns importance to each operation
        \begin{itemize}
			\item Since the $\alpha$ are continuous, we can optimize them with gradient descent
        \end{itemize}
        \pause
	\item In the end, DARTS discretizes to obtain a single architecture 
    \end{itemize}
    
\end{frame}

\begin{frame}{DARTS: Architecture Optimization}

					%\nonumber			
					$\text{min}_{\alpha} \mathcal{L}_{\text{\alert{val}}}(w^*(\alpha), \alert{\alpha})$\\
					%\nonumber			
					$s.t.\ w^*(\alpha) \in \text{argmin}_{w} \mathcal{L}_{\text{\alert{train}}}(\alert{w},\alpha)$

		\end{center}	
		\item This is solved using alternating SGD steps on architectural parameters $\alpha$\\and weights $w$

\begin{frame}{Issues -- Non-Robust Behavior}
\begin{itemize}
    \item DARTS is very sensitive w.r.t. its own hyperparameters (e.g., one-shot learning rate, $L_2$ regularization, etc.)
    \begin{itemize}
        \item Tuning these hyperparameters for every new task/search space is computationally expensive
        \item DARTS may return degenerate architectures, e.g., cells composed with only skip connections
    \end{itemize}
\end{itemize}

\begin{frame}{Issues -- Memory Constraints}
\begin{itemize}
    \item DARTS keeps the \alert{entire one-shot model in memory}, together with its computed tensors 
    \begin{itemize}
        \item  This constrains the search space size and the fidelity used to train the one-shot model
        \item \alert{Impossible} to run on large datasets such as ImageNet
    \end{itemize}
\end{itemize}

\begin{frame}{Issues -- Memory Constraints}
\begin{itemize}
    \item DARTS keeps the \alert{entire one-shot model in memory}, together with its computed tensors 
    \begin{itemize}
        \item  This constrains the search space size and the fidelity used to train the one-shot model
        \item \alert{Impossible} to run on large datasets as ImageNet
    \end{itemize}
\end{itemize}

A lot of research aims to fix this issue:
\begin{itemize}
    \item \alert{GDAS} \liturl{Dong et al, 2019}{https://arxiv.org/pdf/1910.04465.pdf} -- samples from a Gumbel Softmax distribution to keep only a single architecture in memory
    \item \alert{ProxylessNAS} \liturl{Cai et al, 2019}{https://openreview.net/pdf?id=HylVB3AqYm} -- computes approximate gradients on $\alpha$ keeping only 2 edges between two nodes in memory at a time 
    \item \alert{PC-DARTS} \liturl{Xu et al, 2020}{https://arxiv.org/pdf/1907.05737.pdf} -- performs the search on a subset of the channels in the one-shot mode
\end{itemize}
\end{frame}


\begin{frame}{Summary}
    
Most important insights from this video:

\begin{enumerate}
\item DARTS uses continuous architecture weights $\alpha$ to make the search differentiable
\item The bi-level optimization alternates between updating weights $w$ and architecture parameters $\alpha$
\item DARTS is efficient but sensitive to hyperparameters and may produce degenerate architectures
\end{enumerate}
\end{frame}
