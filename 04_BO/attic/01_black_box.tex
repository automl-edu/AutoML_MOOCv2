\begin{frame}{Standard vs.\ Black-Box Optimization}
    \textbf{Optimization:} Find
    $$\min_{\xv \in \mathcal{S}} f(\xv)$$
    with objective function $f: \mathcal{S} \to \R$, where $\mathcal{S}$ is usually box-constrained.
    \vfill
    \begin{columns}
        \column{.4\textwidth}
        \begin{center}
            \includegraphics[width=.95\textwidth]{images/Multimodal-example1.png}
        \end{center}
        \column{.6\textwidth}
        If we are lucky\ldots
        \begin{itemize}
            \item \ldots we have an analytic description of $f: \mathcal{S} \to \R$
            \item \ldots we can calculate gradients and use gradient-based methods (e.g.\ gradient descent)
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Standard vs.\ Black-Box Optimization}
    \textbf{Optimization:} Find
    $$\min_{\xv \in \mathcal{S}} f(\xv)$$
    with objective function $f: \mathcal{S} \to \R$, where $\mathcal{S}$ is usually box-constrained.
    \vfill
    \begin{columns}
        \column{.4\textwidth}
        \begin{center}
            \includegraphics[width=.95\textwidth]{images/cross-in-tray.jpg}
        \end{center}
        \column{.6\textwidth}
        Optimization gets harder\ldots
        \begin{itemize}
            \item \ldots if we cannot calculate gradients (because $f$ is not differentiable or not known)
            \item \ldots but as long as evaluations of $f$ are cheap, we can use standard derivative-free methods (e.g.\ Nelder-Mead, simulated annealing, EAs)
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Naive Approaches}
    \textbf{Empirical knowledge / manual tuning}
    \begin{itemize}
        \item Select parameters based on ``expert'' knowledge
        \item \textbf{Advantages:} Can lead to fairly good outcomes for known problems
        \item \textbf{Disadvantages:} Very inefficient, poor reproducibility, chosen solution can be far away from the global optimum
    \end{itemize}
\end{frame}

%----------------------------------------------------------------------

\begin{frame}{Naive Approaches}
    \textbf{Random search / Grid search}
    \begin{itemize}
        \item Random search: evaluate uniformly sampled inputs
        \item Grid search: exhaustive search on a predefined grid of inputs
        \item \textbf{Advantages:} Easy, intuitive, parallelization trivial
        \item \textbf{Disadvantages:} Inefficient, search large irrelevant areas
    \end{itemize}

\begin{frame}{Naive Approaches}
    \textbf{Traditional black-box optimization}
    \begin{itemize}
        \item Traditional approaches that do not require derivatives
        \item E.g.\ Nelder-Mead, simulated annealing, EAs
        \item \textbf{Advantages:} Truly iterative, focuses on relevant regions
        \item \textbf{Disadvantages:} Still inefficient; usually many evaluations needed to produce good outcomes
    \end{itemize}

\end{frame}

\begin{frame}{Summary}
\begin{enumerate}
    \item Black-box optimization: no analytic form, no gradients, expensive to evaluate
    \item Applications range from robotics to recipe design
    \item Naive approaches (manual, grid, random, classical derivative-free) are intuitive but inefficient in the expensive setting
\end{enumerate}
\end{frame}
