\begin{frame}[t]{Optimization via Surrogate Modeling}
  \begin{itemize}
    \item Want to solve $$\conf^\ast \in \argmin_{\conf \in \pcs} \cost(\conf)$$
    \item We do not know the analytical structure of $\cost: \pcs \to \realnum$;\\
     it is only available as a black-box we can \textbf{query} at points $\conf \in \pcs$
    \item For now, we assume evaluations are noise-free
    \item \textbf{Idea:} Use the archive of past evaluations
    $\hist = \datasetHPOdef$
    to learn as much as possible about $\cost$, and plan subsequent evaluations \textbf{intelligently}
  \end{itemize}
  % \looppic{images/02_bo_loop_1d_toy_1.pdf}
\end{frame}

\begin{frame}{Initial Design}
    We need a first batch of evaluations before any surrogate can be fit.
    \begin{itemize}
        \item Should \textbf{cover / explore} the input space sufficiently so the initial surrogate is not biased toward one region:
            \begin{itemize}
                \item Random design
                \item Latin hypercube sampling (LHS)
                \item Sobol / Halton quasi-random sequences
            \end{itemize}
        \item Type of design usually does not have the largest effect~\liturl{Bossek et al. 2020}{https://arxiv.org/abs/2003.13826}
        \item More important: the \textbf{size} of the initial design
            \begin{itemize}
                \item Too small: bad initial fit, surrogate misleading
                \item Too large: budget wasted before ``intelligent'' optimization even starts
                \item Rule of thumb: $4 \pcsdim$, where $\pcsdim = \dim(\pcs)$
            \end{itemize}
    \end{itemize}
\end{frame}


\begin{frame}{Latin Hypercube Sampling}
    \begin{columns}[onlytextwidth]
        \column{.5\textwidth}
        \begin{itemize}
            \item LHS partitions the configuration space into bins of equal probability
            \item Goal: attain a more even distribution of sample points than random sampling
            \item Allow at most 1 sample per bin; exactly 1 sample per row and column
            \item Either use midpoint or uniform sample from bin
        \end{itemize}
        \column{.5\textwidth}
        \begin{center}
            \includegraphics[width=.5\textwidth]{images/02_initdes_randomvslhs_random.pdf}\\
            \includegraphics[width=.5\textwidth]{images/02_initdes_randomvslhs_lhs.pdf}
            % \footnotesize Marginal histograms: RS vs.\ LHS
        \end{center}
    \end{columns}
\end{frame}

\begin{frame}{Space-Filling Criterion: Maximin}
\begin{columns}[onlytextwidth]
    \column{.75\textwidth}
    \begin{itemize}
    \item Specific LHS design selected via, e.g., \textbf{maximin}
    \item For a given design $\dataset$, measure min.\ pairwise distance for any metric $\rho$:
     $2q = \min_{\conf \neq \conf' \in \dataset} \rho(\conf, \conf')$
    \item $q$ = \textbf{packing radius}: radius of largest ball around each point s.t. no 2 balls overlap
    \item Goal: find $\dataset$ that maximizes $2q$, keeps points far apart:
    $$\max_{\dataset} \min_{\conf \neq \conf' \in \dataset} \rho(\conf, \conf')$$
    \item Combinatorial problem; solved heuristically, e.g., by simulated annealing on 
    column swaps~\liturl{Morris et al. 1995}{https://www.sciencedirect.com/science/article/abs/pii/037837589400035T} or random search
    \end{itemize}
    \column{.25\textwidth}
    \begin{center}
    \begin{tikzpicture}[scale=2.3,
        pt/.style={circle, fill=black, inner sep=1.2pt},
        ball/.style={draw=blue!60, fill=blue!10, thick}]
        \draw[black!40] (0,0) rectangle (1,1);
        % Closest pair is (0.40,0.30)-(0.22,0.45), distance ~0.234,
        % so packing radius q = 0.117 -> circles just touch, never overlap.
        \foreach \x/\y in {0.12/0.82, 0.40/0.30, 0.72/0.68, 0.88/0.15, 0.22/0.45} {
            \draw[ball] (\x,\y) circle (0.117);
        }
        \foreach \x/\y in {0.12/0.82, 0.40/0.30, 0.72/0.68, 0.88/0.15, 0.22/0.45} {
            \node[pt] at (\x,\y) {};
        }
        \draw[red, thick, <->] (0.40,0.30) -- (0.22,0.45)
            node[midway, above right, font=\scriptsize, red] {$2q$};
    \end{tikzpicture}\\[0.2cm]
    % \footnotesize Packing radius $q$ around each point; closest pair determines $2q$.
    \end{center}
\end{columns}
\end{frame}

\begin{frame}[t]{Exploration vs.\ Exploitation}
  The algorithm ran straight into a \textbf{local} minimum -- it did not explore the most crucial areas and \textbf{missed} the global optimum.
  \vspace{0.3cm}
  \begin{columns}[onlytextwidth,T]
      \column{.48\textwidth}
      \includegraphics[width=\textwidth]{images/02_bo_loop_1d_toy_11.pdf}
      \column{.48\textwidth}
      \begin{itemize}
          \item Optimizing $\surro$ directly uses the \textbf{mean prediction} as proposal rule
          \item That is pure \textbf{exploitation}, no \textbf{exploration}
          \item Smarter proposal rules -- \textbf{acquisition functions} -- also reward visiting uncertain regions
      \end{itemize}
  \end{columns}
\end{frame}


\begin{frame}{Summary - The Basic BO Loop}
    \begin{enumerate}
        \item Create \textbf{initial design}
        \item \textbf{Fit} surrogate model $\surro$ on the archive $\hist = \datasetHPOdef$
        \item \textbf{Propose} new point $\confI{t+1}$ by optimizing acquisition function $\acq(\conf)$
        \item \textbf{Evaluate} $\confI{t+1}$ and \textbf{update} the archive:\\ $\hist \leftarrow \hist \cup \langle \confI{t+1}, \cost(\confI{t+1}) \rangle$
        \item Repeat from step 2. until budget is exhausted
    \end{enumerate}
    \vspace{0.2cm}
    Core design choices: the \textbf{initial design}, the \textbf{surrogate model}, and the \textbf{rule} used to propose the next point.
    % \vspace{0.2cm}
    %
    % \textbf{Further reading:} \liturl{Forrester et al. 2008}{https://onlinelibrary.wiley.com/doi/book/10.1002/9780470770801} -- classic surrogate-modelling textbook with full LHS / maximin / BO treatment.
\end{frame}
