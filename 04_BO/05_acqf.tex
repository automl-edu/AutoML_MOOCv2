\begin{frame}[t]{Bayesian Surrogate Modeling}
  \small
  \begin{itemize}
    \item \textbf{Goal:} Find a trade-off between \textbf{exploration} (areas we have not visited yet) and \textbf{exploitation} (search around good design points)
    \item \textbf{Idea:} Use a \textbf{Bayesian approach} to build a surrogate model yielding estimates for the posterior mean $\surro(\conf)$ and posterior variance $\sh^2(\conf)$
    \item $\sh^2(\conf)$ expresses ``confidence''/``certainty'' in the prediction
    % \item To sequentially propose new points based on the surrogate model, we use \textbf{acquisition functions} $\acq: \pcs \to \realnum$
    % \item Let $\cmin := \min\{\cost(\confI{1}), \ldots, \cost(\confI{t})\}$ be the best observed value so far (green in plots)
    % \item In the examples before, we simply used the posterior mean $\acq(\conf) = \surro(\conf)$ as acquisition function -- ignoring uncertainty
  \end{itemize}

\begin{frame}[t]{Lower Confidence Bound}
% \begin{columns}[onlytextwidth,T]
%   \column{.62\textwidth}
  \begin{itemize}
    \item Minimize $\acq_{\text{LCB}}(\conf) = \surro(\conf) - \tau \sh(\conf)$
    \item $\tau > 0$ controls the ``mean vs.\ uncertainty'' trade-off
    \item Very simple, does not rely on distributional assumptions
    \item Regret bounds for GP-UCB / LCB~\liturl{Srinivas et al. 2010}{https://arxiv.org/abs/0912.3995}
  \end{itemize}

\begin{frame}[t]{Probability of Improvement}
  \footnotesize
  \begin{itemize}
    \item Maximize $\acq_{\text{PI}}(\conf) = \PP(Y(\conf) < \cmin) = \Phi\!\left(\frac{\cmin - \surro(\conf)}{\sh(\conf)}\right)$
    \item $\Phi(\cdot)$ is the standard normal cdf (assuming Gaussian posterior)
    \item PI does not take the size of the improvement into account
    \item Often proposes points close to the current $\cmin$
    \item $\acq_{\text{PI}}(\confI{i}) = 0$ at design points $\confI{i}$:\\
    $\sh(\confI{i}) = 0$ for an interpolating GP $\Rightarrow Y(\confI{i})$ is degenerate at $\cost(\confI{i})$,\\
    so $\PP(Y(\confI{i}) < \cmin) = \ind[\cost(\confI{i}) < \cmin] = 0$ since $\cost(\confI{i}) \ge \cmin$
  \end{itemize}

\begin{frame}[t]{Expected Improvement}
  \begin{itemize}
    \item Maximize $\acq_{\text{EI}}(\conf) = \E(\max\{\cmin - Y(\conf), 0\})$
    \item We take the \textbf{expectation} in the tail, instead of the probability as in PI
    \item Improvement is always assumed $\geq 0$, enforces exploration
    \item Originally introduced by~\liturl{Mo\v{c}kus 1975}{https://link.springer.com/chapter/10.1007/978-3-662-38527-2_55};\\
    popularized for expensive simulators by~\liturl{Jones et al. 1998}{https://link.springer.com/article/10.1023/A:1008306431147} (EGO)
  \end{itemize}

\begin{frame}[t]{Expected Improvement}
  \begin{itemize}
    \item If $Y(\conf) \sim \normaldist\left(\surro(\conf), \sh^2(\conf)\right)$, the EI has a closed form:
    $$\acq_{\text{EI}}(\conf) = (\cmin-\surro(\conf))\, \Phi\!\left(\frac{\cmin - \surro(\conf)}{\sh(\conf)}\right) + \sh(\conf)\, \phi\!\left(\frac{\cmin - \surro(\conf)}{\sh(\conf)}\right)$$
    \item Can easily show this by partial integration
    \item $\acq_{\text{EI}}(\confI{i}) = 0$ at design points $\confI{i}$:\\
    $\sh(\confI{i}) = 0$ for an interpolating GP $\Rightarrow Y(\confI{i})$ is degenerate at $\cost(\confI{i})$,\\
    so $\acq_{\text{EI}}(\confI{i}) = \max(\cmin - \cost(\confI{i}),\, 0) = 0$ since $\cost(\confI{i}) \ge \cmin$
  \end{itemize}
\end{frame}

\begin{frame}[t]{Summary and Discussion}
  \begin{itemize}
    \item ACQFs balance exploitation and exploration,\\
    usually by combining posterior mean and uncertainty
    \item 3 basic ACQFs: PI, LCB, EI
    \item EI is most standard; LCB also works well in practice
    \item Under some mild conditions: BO with a GP and EI is a \textbf{global optimizer}, i.e., convergence to the \textbf{global} optimum is guaranteed given unlimited budget~\liturl{Bull 2011}{https://www.jmlr.org/papers/volume12/bull11a/bull11a.pdf}
    \item Other acquisition functions: entropy search, predictive entropy search~\liturl{Hern\'andez-Lobato et al. 2014}{https://arxiv.org/abs/1406.2541}, max-value entropy search, knowledge gradient, Thompson sampling, \ldots
  \end{itemize}
\end{frame}
