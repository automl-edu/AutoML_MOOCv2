\begin{frame}[t]{Acquisition Function Properties}
  \small
  \begin{itemize}
    \item In each BO loop, we exchange the optimization of $\cost(\conf)$ for the acquisition optimization
    of $\acq(\conf)$
    \item The 2 optimization problems have many similarities, but also differences
    \item The $\pcs$ configuration space structure is the same
    \item The objective structure $\acq(\conf)$ is somehow shaped by the type of ACQF but likely 
    complicated and multi-modal
    \item But problem now cheap since we query the model $\surro(\conf)$ instead of the costly $\cost(\conf)$
    \item NB: We do not have to solve this problem perfectly, as the surrogate is only a coarse approximation
  \end{itemize}
  \vspace{0.2cm}
  \begin{center}
  \textbf{We still face a non-trivial, multi-modal problem -- maybe differentiable, maybe on complicated 
  configuration space structure}
  \end{center}
\end{frame}


\begin{frame}[t]{The Acquisition Function Landscape}
  \small
  \begin{itemize}
    \item $\acq$ is typically \textbf{non-convex} and \textbf{multi-modal}
    \item For EI / PI: $\acq(\conf) = 0$ at training points; peaks sit between them
    \item Landscape gets more rugged as the surrogate is fit to more data
    \item In high dimensions: flat plateaus with narrow peaks -- hard to find
  \end{itemize}

\begin{frame}[t]{ACQFs on Log-Scale}
  % Plain EI underflows to a hard $0$ in well-explored regions ($\hat\mean(\conf) \gg \cmin$ or tiny $\sh$) $\Rightarrow$ flat plateaus, vanishing gradients, L-BFGS-B stalls. Two distinct cures both go by the name ``LogEI'':
  \begin{itemize}
    \item \textbf{LogEI for numerical stability}~\liturl{Ament et al. 2023}{https://arxiv.org/abs/2310.20708}    
    \begin{itemize}
    \item Compute $\log\mathrm{EI}(\conf)$ directly via stable log-space arithmetic (\texttt{erfcx}, \texttt{log1p})
    \item Simple monotone transform $\Rightarrow$ same argmax as EI
    \item Available trick in BoTorch for many ACQFs
    \end{itemize}
      \item \textbf{LogEI on log-transformed targets}~\liturl{Hutter et al. 2009}{https://dl.acm.org/doi/10.1145/1569901.1569940}: 
      \begin{itemize}
        \item Train the surrogate on $\log \cost$
        \item Helps when $y$ is heavily right-tailed
        \item Then EI \textbf{on the original $y$-scale} has the log-normal closed form
    $$\acq_{\mathrm{LogEI}}(\conf) = f_{\min}\,\Phi(v) - \exp\!\big(\mu + \tfrac{1}{2}\sigma^2\big)\,\Phi(v - \sigma), \quad v = (\log f_{\min} - \mu)/\sigma$$
    \item Often good results in practice
  \end{itemize}
  \end{itemize}
\end{frame}


\begin{frame}[t]{Numeric Configuration Spaces}
\begin{itemize}
\item \textbf{Random search}
\begin{itemize}
\item Some people do this due to simplicity and the ``we do not have to optimize perfectly'' point
\item Likely not a good idea in higher dimensions
\end{itemize}
\item \textbf{Multi-start local (gradient-based) search:}
\begin{itemize}
\item Sample multiple start points
\item Run multiple gradient-based local optimizers from each, e.g., L-BFGS-B 
\item Analytic $\nabla \acq$ is available for GPs with numeric inputs, standard kernels, and standard ACQFs
\item Works better with GPs than RFs, and if we have gradients
\end{itemize}
\end{itemize}
\end{frame}

  % \end{itemize}
  \textbf{CMA-ES -- covariance matrix adaptation evolution strategy}
  \begin{columns}[onlytextwidth,T]
    \column{.6\textwidth}
    \begin{itemize}
    \item Population-based, gradient-free
    \item Each generation: sample candidates from $\mathcal{N}(\bm{m}, \sigma^2 \bm{C})$, 
    rank by fitness, update mean $\bm{m}$, step size $\sigma$, and covariance $\bm{C}$ from best candidates
    \item $\bm{C}$ learns the local curvature\\
    $\leadsto$ robust on non-convex, ill-conditioned functions
    \item But: once $\sigma$ contracts around a basin, the search is effectively local -- a single run can miss better modes
    \item Restarts are standard on multi-modal functions:
      \begin{itemize}
      \item IPOP-CMA-ES~\liturl{Auger et al. 2005}{https://ieeexplore.ieee.org/document/1554902} %-- doubles $\lambda$ on every restart to widen exploration
      \item BIPOP-CMA-ES~\liturl{Hansen 2009}{https://dl.acm.org/doi/10.1145/1570256.1570333} %-- interleaves IPOP-like restarts with small-$\lambda$, randomized-$\sigma$ restarts; winner of multiple BBOB benchmarks
      \end{itemize}
    \end{itemize}
    \column{.4\textwidth}
    \begin{center}
      \includegraphics[width=\textwidth]{images/06_cmaes_sampling.pdf}\\[0.1cm]
      \scriptsize 2 generations on a tilted valley:\\
      $\times$ = mean $\bm{m}$, dashed = $2\sigma$ ellipse of $\sigma^2 \bm{C}$,\\
      filled $\bullet$ = top $\mu$, open $\circ$ = the rest
    \end{center}
  \end{columns}
  \end{frame}
  


\begin{frame}[t]{Mixed / Hierarchical Configuration Spaces}
  Iterated one-exchange local search~\liturl{Lindauer et al. 2022}{https://arxiv.org/abs/2109.09831}
  \begin{columns}[onlytextwidth,T]
    \column{.6\textwidth}
    \begin{itemize}
      \item Run multiple local searches in parallel
      \item Generate multiple neighbors by changing \textbf{exactly one} HP:
        \begin{itemize}
          \item Categorical: try other level
          \item Numeric / integer: small (Gaussian) perturbation around the current value
          \item Conditional HPs: only mutate children of \textbf{active} parents
        \end{itemize}
      \item Move to best-improving neighbor
      \item \textbf{Restart} from a fresh random point if no improvement for some iterations
    \end{itemize}
    \column{.4\textwidth}
    \begin{center}
    \begin{tikzpicture}[scale=0.9, font=\scriptsize]
      \draw[gray!25] (-0.2, -0.1) ellipse (2.4 and 1.6);
      \draw[gray!40] (-0.2, -0.1) ellipse (1.5 and 1.0);
      \draw[gray!55] (-0.2, -0.1) ellipse (0.7 and 0.45);
      % faint axes through lambda to make the one-HP-change geometry explicit
      \draw[gray!50, very thin] (-1.7, 0) -- (1.7, 0)
            node[anchor=west, xshift=2pt, color=gray!70] {$\lambda_1$};
      \draw[gray!50, very thin] (0, -1.3) -- (0, 1.3)
            node[anchor=south, color=gray!70] {$\lambda_2$};
      \filldraw[black] (0, 0) circle (3pt);
      \node[anchor=north east, xshift=-2pt, yshift=2pt] at (0, 0) {$\lambda$};
      % each neighbor changes exactly one coordinate of lambda -- so they all
      % lie on the axes through lambda, not on diagonals
      \draw[gray, dashed, ->] (0, 0) -- (-1.05, 0);
      \filldraw[white, draw=black, thick] (-1.05, 0) circle (3pt);
      \draw[gray, dashed, ->] (0, 0) -- (0,  1.2);
      \filldraw[white, draw=black, thick] (0,  1.2) circle (3pt);
      \draw[gray, dashed, ->] (0, 0) -- (0, -0.65);
      \filldraw[white, draw=black, thick] (0, -0.65) circle (3pt);
      % best-improving neighbor: a +lambda_1 step
      \draw[mygreen!55!black, thick, ->] (0, 0) -- (1.55, 0);
      \filldraw[mygreen, draw=mygreen!55!black, thick] (1.55, 0) circle (3pt);

\end{frame}


\begin{frame}[t]{Summary and Discussion}
  \begin{itemize}
    \item Proposing a point via ACQF is itself an optimization problem;
    but based on the \textbf{cheap} surrogate
    \item $\acq$ is non-convex, multi-modal, and gets rugged with more data
    \item Numeric $\pcs$: multi-start L-BFGS-B (w/ GP gradients)\\
    CMA-ES (+ IPOP/BIPOP restarts)
    \item Mixed / hierarchical $\pcs$: SMAC-style iterated local search 
    \item It is unclear how perfectly we have to solve the ACQF optimization problem
    and how much (overhead) budget we should spend on this
  \end{itemize}
