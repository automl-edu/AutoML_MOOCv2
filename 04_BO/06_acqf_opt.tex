\begin{frame}[t]{Acquisition Function Properties}
  \small
  \begin{itemize}
    \item In each BO loop, we exchange the optimization of $\cost(\conf)$ to the acquisition optimization
    of $\acq(\conf)$
    \item The 2 optimization problems have many similarities, but also differences
    \item The $\pcs$ search space structure is the same
    \item The objective structure $\acq(\conf)$ is somehow changed by the type of ACQF but likely 
    complicated and multi-modal
    \item But the problem now becomes cheap
    \item NB: We do not have to solve this problem perfectly, as the surrogate is only a coarse approximation
  \end{itemize}
  \vspace{0.2cm}
  \begin{center}
  \textbf{We still face a non-trivial, multi-modal problem -- maybe differentiable, maybe on complicated 
  search space structure}
  \end{center}
\end{frame}


\begin{frame}[t]{The Acquisition Function Landscape}
  \small
  \begin{itemize}
    \item $\acq$ is typically \textbf{non-convex} and \textbf{multi-modal}
    \item For EI / PI: $\acq(\conf) = 0$ at training points; peaks sit between them
    \item Landscape gets more rugged as the surrogate is fit to more data
    \item In high dimensions: flat plateaus with narrow peaks -- hard to find
  \end{itemize}

\begin{frame}[t]{Numeric Search Spaces}
\begin{itemize}
\item \textbf{Random search}
\begin{itemize}
\item Some people do this due to simplicity and the "we do not have to optimize perfectly" point
\item Likely not a good idea in higher dimensions
\end{itemize}
\item \textbf{Multi-Start local search:}
\begin{itemize}
\item Sample multiple start points
\item Run multiple gradient-based local optimizer from each (L-BFGS-B if $\pcs$ is box-constrained)
\item Depending on what we do exactly, we might have analytic gradients available for $\acq(\conf)$,
e.g. GPs only with numeric inputs, with usual kernels and standard ACQFs allow this 
\item Works better with GPs than RFs, and if we have gradients
\end{itemize}
\end{itemize}
\end{frame}


\begin{frame}[t]{Numeric Search Spaces}
  \small
  \begin{itemize}
  % \item \textbf{DIRECT} (DIviding RECTangles):

  % \item Per iteration: mark ``potentially optimal'' rectangles -- those that could hold a better point under \emph{some} Lipschitz constant -- and trisect them along their longest side
  % \end{itemize}
  \item \textbf{CMA-ES -- Covariance Matrix Adaptation Evolution Strategy}
  \begin{itemize}
  \item Population-based, gradient-free
  \item Each generation: sample $\lambda$ candidates from $\mathcal{N}(\bm{m}, \sigma^2 \bm{C})$, rank by $\acq$, update mean $\bm{m}$, step size $\sigma$, and covariance $\bm{C}$ from the best $\mu$
  \item $\bm{C}$ learns the local curvature\\
  $\leadsto$ robust on non-convex, ill-conditioned $\acq$; handles medium $d$ well
  \item But: once $\sigma$ contracts around a basin, the search is effectively \emph{local} -- a single run can miss better modes elsewhere
  \item Restarts are standard on multi-modal $\acq$; dedicated schemes exist:
  \begin{itemize}
  \item \textbf{IPOP-CMA-ES} (Auger \& Hansen, 2005) -- doubles $\lambda$ on every restart to widen exploration
  \item \textbf{BIPOP-CMA-ES} (Hansen, 2009) -- interleaves IPOP-like restarts with small-$\lambda$, randomised-$\sigma$ restarts; winner of multiple BBOB benchmarks
  \end{itemize}
  \end{itemize}
  \end{itemize}
  \end{frame}
  


\begin{frame}[t]{Mixed / Hierarchical Search Spaces}
  \small
  \begin{itemize}
    \item AutoML spaces mix continuous, integer, categorical, and \emph{conditional} HPs
    \item No gradients and no smoothness across categorical cuts $\leadsto$ L-BFGS / DIRECT / CMA-ES don't apply out-of-the-box
    \item \textbf{SMAC's approach:} interleaved random search + \emph{iterated local search} on $\acq$ (Hutter et al., LION 2011)
  \end{itemize}
  \vspace{0.15cm}
  \textbf{One-exchange local search from each start point:}
  \begin{enumerate}\setlength\itemsep{0pt}
    \item Start from the current incumbent (plus the top-$k$ archive entries)
    \item Generate neighbours by changing \emph{exactly one} HP:
      \begin{itemize}\setlength\itemsep{0pt}
        \item Categorical: try each other level
        \item Numeric / integer: small (Gaussian) perturbation around the current value
        \item Conditional HPs: only mutate children of \emph{active} parents
      \end{itemize}
    \item Move to the best-improving neighbour; stop when no neighbour improves $\acq$
  \end{enumerate}
  \vspace{0.1cm}
  \textbf{Interleave with random search to avoid being stuck in local optima:}
  \begin{itemize}\setlength\itemsep{0pt}
    \item Draw a large pool of random configs, score under $\acq$
    \item Return the best across local-search optima and random samples
  \end{itemize}
\end{frame}


\begin{frame}[t]{Summary and Discussion}
  \small
  \textbf{Key takeaways:}
  \begin{itemize}\setlength\itemsep{0pt}
    \item BO's inner loop is itself an optimization problem -- but over the \emph{cheap} surrogate; we don't need to solve it perfectly
    \item $\acq$ is non-convex, multi-modal, and gets rugged with more data
    \item Numeric $\pcs$: multi-start L-BFGS-B (w/ GP gradients), DIRECT, or CMA-ES (+ IPOP/BIPOP restarts)
    \item Mixed / hierarchical $\pcs$: SMAC-style iterated local search interleaved with random search
  \end{itemize}
  \vspace{0.1cm}
  \textbf{Budget: \#starts $\times$ iterations per start -- scales with $d$}
  \begin{itemize}\setlength\itemsep{0pt}
    \item Too few restarts $\leadsto$ miss the global peak, propose suboptimal points
    \item Too many $\leadsto$ per-iteration overhead dominates, especially when $\cost$ is cheap
    \item Typical defaults: $10$--$50$ restarts for GP + L-BFGS-B in $\le 20$ dimensions
  \end{itemize}
  \vspace{0.1cm}
  \textbf{When inner optimization matters most (invest CPU):}
  \begin{itemize}\setlength\itemsep{0pt}
    \item Expensive $\cost$ (minutes-to-hours per eval), late in the BO run (AF peaks narrow), noisy / multi-objective $\acq$
    \item When \emph{not} to over-engineer: cheap $\cost$ with many BO iterations; low-$d$, smooth surrogate ($\sim\!5$ L-BFGS-B restarts suffice)
  \end{itemize}
\end{frame}
